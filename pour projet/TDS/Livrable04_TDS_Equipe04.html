from scipy.signal import butter, filtfilt                                                    # permet de faire le filtre passe bas

def filtre_passe_bas(signal, frequence_coupure=300, frequence_echantillonnage=Fe, ordre=5):  #filtre passe bas (pour la dmodulation ASK)
    nyquist = 0.5 * frequence_echantillonnage
    normalise_coupure = frequence_coupure / nyquist
    b, a = butter(ordre, normalise_coupure, btype='low', analog=False)
    return filtfilt(b, a, signal)

#Modulation
M = []  # Message binaire encod M
for element in binaire_encoder:
    M.append(int(element))
    
Fe = 44100  # Frquence d'chantillonnage
Fp = 20000  # Frquence de l'onde porteuse
baud = 300  # Dbit souhait en bit/s


Nbits = len(M)  # Nombre de bits
Ns = int(Fe / baud)  # Nombre d'chantillons par bit
N = Ns * Nbits  # Nombre total d'chantillons

# Gnration du message binaire dupliqu
M_duplique = np.repeat(M, Ns)

# Gnration du vecteur temps
t = np.arange(N) / Fe  # Assure une correspondance exacte avec M_duplique

# Gnration de la porteuse
P = np.cos(2 * np.pi * Fp * t)  # Porteuse unique

# Modulation ASK
ASK = M_duplique * P

# Affichage des rsultats
plt.figure(figsize=(10, 6))  #on dfinit la taille des graphiques

# Affichage du message binaire
plt.subplot(3, 1, 1)
plt.plot(t[:Ns*5], M_duplique[:Ns*5], 'k')        #ici nous n'affichons que les 5 premiers bits sur les 3 graphiques pour plus de lisibilit
plt.title("Message Binaire")
plt.ylim(-0.2, 1.2)

# Affichage de la porteuse
plt.subplot(3, 1, 2)
plt.plot(t[:Ns*5], P[:Ns*5], label="Porteuse")
plt.title("Porteuse")
plt.legend()

# Affichage du signal ASK
plt.subplot(3, 1, 3)
plt.plot(t[:Ns*5], ASK[:Ns*5])
plt.title("Modulation ASK")

plt.tight_layout()
plt.show()
